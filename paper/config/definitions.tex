% source: http://www.maths.tcd.ie/~dwilkins/LaTeXPrimer/Theorems.html
% See also https://en.wikibooks.org/wiki/LaTeX/Theorems

% Counting at chapter level. All have same counter.
\theoremstyle{plain}
\newtheorem{theorem}{Theorem}[section]
\newtheorem{lemma}[theorem]{Lemma}
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{corollary}[theorem]{Corollary}

% Assumption has its own counter.
\newtheorem{assumption}{Assumption}[section]

% Defintions switch to normal font, not italicised.
\theoremstyle{definition} 
\newtheorem{definition}{Definition}[section]
\newtheorem{algorithm}{Algorithm}[section]

% Examples and remarks are not numbered.
\theoremstyle{remark}
\newtheorem*{example}{Example}
\newtheorem*{remark}{Remark}
